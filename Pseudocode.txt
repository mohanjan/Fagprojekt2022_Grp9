FIR_LPF(S,L){
    Z[0]=S
    sum=FIR_C(Z[0],0)
    for i=1...L{
        Z[i]=Z[i-1]
        sum+=FIR_C(Z[i],i)
    }
   return sum
}

FIR_C(S,c){
    return S*C[c]
